ProbablyFresh: full start from zero (PowerShell, Docker only)

1) Open project folder
cd F:\DE_intern\probablyfresh-analytics-platform

2) Prepare env (one time)
copy .env.example .env

3) Build app image (one time, or after requirements/Dockerfile changes)
docker compose --env-file .env build app

4) Full reset of infrastructure and data
docker compose --env-file .env down -v

5) Start all required services
docker compose --env-file .env up -d zookeeper kafka mongodb clickhouse app grafana

6) Wait until ClickHouse is ready (important)
do { Start-Sleep -Seconds 2; docker compose --env-file .env exec -T clickhouse clickhouse-client --query "SELECT 1" } while ($LASTEXITCODE -ne 0)

7) Initialize ClickHouse RAW + MART
Get-Content docker/clickhouse/init/01_init.sql -Raw | docker compose --env-file .env exec -T clickhouse clickhouse-client --multiquery
Get-Content docker/clickhouse/init/02_mart.sql -Raw | docker compose --env-file .env exec -T clickhouse clickhouse-client --multiquery

8) Generate demo JSON data
docker compose --env-file .env run --rm app python src/generator/generate_data.py

9) Load JSON into MongoDB
docker compose --env-file .env run --rm app python src/loader/load_to_mongo.py

10) Publish Mongo -> Kafka (once)
docker compose --env-file .env run --rm app python src/streaming/produce_from_mongo.py --once

11) Give ClickHouse time to consume Kafka messages
Start-Sleep -Seconds 5

12) Refresh quality snapshot (from 02_mart.sql)
Get-Content docker/clickhouse/init/02_mart.sql -Raw | docker compose --env-file .env exec -T clickhouse clickhouse-client --multiquery

13) Validate RAW/MART row counts
docker compose --env-file .env exec -T clickhouse clickhouse-client --query "SELECT 'stores_raw' AS t, count() AS c FROM probablyfresh_raw.stores_raw UNION ALL SELECT 'products_raw', count() FROM probablyfresh_raw.products_raw UNION ALL SELECT 'customers_raw', count() FROM probablyfresh_raw.customers_raw UNION ALL SELECT 'purchases_raw', count() FROM probablyfresh_raw.purchases_raw"
docker compose --env-file .env exec -T clickhouse clickhouse-client --query "SELECT 'customers_mart' AS t, count() AS c FROM probablyfresh_mart.customers_mart UNION ALL SELECT 'purchases_mart', count() FROM probablyfresh_mart.purchases_mart"

14) Run ETL to S3/MinIO (no repeated pyspark install/download)
docker compose --env-file .env run --rm app spark-submit --master local[*] --jars /opt/jars/clickhouse-jdbc-0.9.6-all-dependencies.jar jobs/features_etl.py

15) Validate output file in MinIO/S3 via mc
mc alias set local http://localhost:9000 $env:MINIO_ACCESS_KEY $env:MINIO_SECRET_KEY
mc ls local/$env:MINIO_BUCKET
mc cat local/$env:MINIO_BUCKET/analytic_result_YYYY_MM_DD.csv | Select-Object -First 10
